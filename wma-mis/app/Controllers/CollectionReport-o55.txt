    public function getVariable($var)
    {
        $input  = $this->appRequest->getVar($var, FILTER_SANITIZE_SPECIAL_CHARS);
        if ($input === null) {
            return '';
        } else {
            return $input;
        }
    }

    public function downloadReportData($selectedCenter, $activity, $task, $dateFrom, $dateTo, $month, $year, $paymentStatus, $title)
    {

        $data = [
            'wma_bill.CollectionCenter' => $this->user->inGroup('officer', 'manager') ? $this->collectionCenter : $selectedCenter,
            'wma_bill.IsCancelled' => 'No',
            'Activity' => $activity,
            'wma_bill.Task' => $task,
            'DATE(wma_bill.CreatedAt)>=' => $dateFrom,
            'DATE(wma_bill.CreatedAt)<=' => str_replace('23 59 59', '23:59:59',$dateTo),
            'MONTH(wma_bill.CreatedAt)' => $month,
            'YEAR(wma_bill.CreatedAt)' => $year,
            'PaymentStatus' => $paymentStatus,
            'PayCntrNum !=' => ''


        ];

        $clearedPartial = 0;

   
        // Printer($data);
      
        // exit;
        // if ($activity == 'others') unset($data['Activity']);
        $params = array_filter($data, fn ($param) => $param !== 'wma');
        $collection = array_map(function ($data) use ($activity) {
            $report = $data;
            $report->billItems = $this->billModel->getBillItems($data->BillItemRef);
            return $report;
        }, $this->billModel->getReportData(params: $params));

        // Printer($data);
        // Printer($params);

        // Printer($collection);
        // exit;



        $output['title'] = humanize($title);
        $output['template'] = $activity == 'wma' ? $this->reportLibrary->reportTemplateAllActivities($collection, $clearedPartial) : $this->reportLibrary->reportTemplate($collection, $clearedPartial);
        // $dsd['template'] = $this->reportLibrary->reportTemplateAllActivities($collection);


        //rendering a pdf using Mpdf 
        $this->pdfLibrary->renderPdf(orientation: 'P', view: 'ReportTemplates/reportView', data: $output, title: $title);
    }
