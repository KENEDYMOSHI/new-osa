    public function reducer($data)
    {

            return  number_format(array_reduce($data, fn ($x, $y) => $x + $y));
        } else {
            return 0;
        }
    }

    public function processPaidItems($params)
    {


        $db = \Config\Database::connect();


        $reportFullPaid = $db->table('bill_payment')->select('
               wma_bill.BillId,
               ItemName,
               GfsCode as Activity,
               CollectionCenter as region,
               bill_payment.id,
               bill_payment.BillAmt,
               PaidAmt,
               PayCtrNum,
               clearedAmount,
               PaidAmount,
               BillItemAmt as amount,
               bill_payment.BillPayOpt,
               PaymentStatus,
               TrxDtTm,
               bill_payment.PyrName,
               bill_payment.CreatedAt

              ')
            ->where($params)
            ->where('bill_payment.BillPayOpt', '3')
            ->limit(0)
            ->join('wma_bill', 'bill_payment.PayCtrNum = wma_bill.PayCntrNum')
            ->join('bill_items', 'bill_payment.BillId = bill_items.BillId')
            ->get()
            ->getResult();




         $activity = $params['Activity'] ?? '';
       // $activity = setting('Gfs.waterMeter');



        unset($params['Activity']);
        $reportPartialPaid = $db->table('bill_payment')->select('
            wma_bill.BillId,
            CollectionCenter as region,
            bill_payment.id,
            bill_payment.BillAmt,
            PaidAmt,
            PayCtrNum,
            clearedAmount,
            bill_payment.BillPayOpt,
            TrxDtTm,
            bill_payment.PyrName,
            bill_payment.CreatedAt
    
           ')
            ->where($params)
            ->where('bill_payment.BillPayOpt', '2')
            ->join('wma_bill', 'bill_payment.PayCtrNum = wma_bill.PayCntrNum')
            ->get()
            ->getResult();





        // Grouping by PayCtrNum
        $grouped = [];
        foreach ($reportPartialPaid as $object) {
            $grouped[$object->PayCtrNum][] = $object;
        }
        $combined = [];
        foreach ($grouped as $payCtrNum => $objects) {
            $combinedPaidAmt = 0;
            $maxBillId = null;
            $maxId = 0;
            foreach ($objects as $object) {
                $combinedPaidAmt += $object->PaidAmt;

                if ($object->id > $maxId) {
                    $maxBillId = $object->BillId;
                    $maxId = $object->id;
                }
            }
             $items =  $db->table('bill_items')->select('ItemName,BillItemAmt,GfsCode as Activity,ItemQuantity')->where('BillId', $maxBillId)->get()->getResult();

        
            $complete = $combinedPaidAmt == $object->BillAmt ? 'Yes' : 'No';
            $itemCount = count($items);
            $strippedAmount = $combinedPaidAmt / $itemCount;

            $combinedItem =
               
                array_map(function ($item) use ($complete, $strippedAmount, $object, $payCtrNum) {
                    $item->customer = $object->PyrName;
                    $item->controlNumber = $payCtrNum;
                    $item->region = $object->region;
                    $item->isFullPaid = $complete;
                    $item->amount = $complete == 'Yes' ? $item->BillItemAmt : $strippedAmount;
                    $item->paymentStatus = 'Paid';
                    return $item;
                }, $items);
            // 'combined' => count($objects) > 1 ? 'Yes' : 'No' // Adding combined key
            array_push($combined, $combinedItem);
        }

        $sortedItems = [];

        foreach ($combined as $key => $value) {
            $innerItem = $value;

            foreach ($innerItem as $item) {
                $sortedItems[] = $item;
            }
        }


        if ($activity != '') {
            $partialPayments = array_filter($sortedItems, fn ($item) => $item->Activity == $activity);
        }else{
            $partialPayments = $sortedItems;
        }


        return array_merge($partialPayments, $reportFullPaid);
    }
